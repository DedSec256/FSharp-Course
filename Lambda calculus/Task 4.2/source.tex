		\textbf{Task 4.2:\\*S K K = I Proof}
	\end{center}

	$\\*S\,K\,K = (\lambda x\,y\,z. x \, z \, (y \, z)) \, K \, K \to
	 \\* \to (\lambda y\,z.K\,z\,(y\,z))\,K
	 \\* \to (\lambda y\,z.(\lambda x\,y.x)\,z\,(y\,z))\,K \to
	 \\* \to (\lambda y\,z.(\lambda y.z)\,(y\,z))\,K
	 \\* \to (\lambda y\,z.z)\,K
	 \\* \to \lambda z.z = I$
	
	\begin{flushleft}
		\textbf{Proved.}
	\end{flushleft}
